from itertools import product
def triangle(a, b, c):
    return a + b > c
perimeter = 21
lengths = range(1, perimeter-1)
count = 0
for a, b in product(lengths,repeat=2):
    c = perimeter - (a + b)
    if a < b and b < c:
        if triangle(a,b,c):
            count += 1
print "Answer:", count
